@pragma("vm:prefer-inline")
String collateListString(List<String> str, [String separate = "\n"]) {
  StringBuffer buff = StringBuffer();
  for (String r in str) {
    buff.write(r);
    buff.write(separate);
  }
  return buff.toString();
}
